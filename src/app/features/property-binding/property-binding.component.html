  <!-- contentEditable binding -->

üìö Property Binding Syntax

Basic Syntax

  <!-- Square brackets syntax -->
  <element [property]="expression">
  
  <!-- Examples -->
  <div [contentEditable]="isEditable"></div>
  <img [src]="imageUrl" [alt]="imageAlt">
  <button [disabled]="isDisabled"></button>

Style Binding

  <!-- Individual styles -->
  <div [style.color]="textColor"></div>
  <div [style.font-size.px]="fontSize"></div>
  
  <!-- Multiple styles -->
  <div [style]="buttonStyles"></div>
  
  <!-- In TypeScript -->
  buttonStyles = {{ '{' }}
    'background-color': buttonColor,
    'color': 'white'
  {{ '}' }};

Class Binding

  <!-- Single class -->
  <div [class.active]="isActive"></div>
  
  <!-- Multiple classes -->
  <div [class]="getButtonClasses()"></div>
  
  <!-- Conditional classes -->
  <div [class.bg-red-500]="hasError"></div>

Attribute Binding

  <!-- Standard attributes -->
  <a [href]="url" [target]="target"></a>
  <input [value]="text" 
  [placeholder]="hint">
  
  <!-- ARIA attributes -->
  <button [attr.aria-label]="buttonLabel">
  </button>
  
  <!-- Data attributes -->
  <div [attr.data-id]="itemId"></div>

üéØ Property vs Attribute

Property Binding

  <!-- Binds to DOM property -->
  <input [value]="username">
  <button [disabled]="isDisabled">
  
  <!-- Updates immediately -->
  <!-- Handles booleans properly -->
  <!-- TypeScript aware -->

Binds to DOM object properties, not HTML attributes

Attribute Binding

  <!-- Binds to HTML attribute -->
  <td [attr.colspan]="colspan">
  <div [attr.data-id]="itemId">
  
  <!-- Use for non-standard attributes -->
  <!-- ARIA attributes -->
  <!-- Data attributes -->

Use [attr.] prefix for HTML attributes
