üíª How Interpolation Works

TypeScript Class Properties

  export class CityComponent {{ '{' }}
    // Properties for interpolation
    city = 'San Francisco';
    temperature = 65;
    isSunny = true;
    
    // Method that can be called
    get weatherMessage() {{ '{' }}
      return this.isSunny 
        ? 'Sunny day!' 
        : 'Cloudy day';
    {{ '}' }}
    
    // Array of data
    landmarks = [
      {{ '{' }} name: 'Golden Gate', visited: true {{ '}' }},
      {{ '{' }} name: 'Alcatraz', visited: false {{ '}' }}
    ];
  {{ '}' }}

HTML Template with Interpolation

  <!-- Display simple property -->
  <p>Hello {{ '{{ city }}' }}</p>
  
  <!-- Math expression -->
  <p>{{ '{{ 1 + 1 }}' }} = 2</p>
  
  <!-- Method call -->
  <p>{{ '{{ weatherMessage }}' }}</p>
  
  <!-- Object property -->
  <p>{{ '{{ landmarks[0].name }}' }}</p>
  
  <!-- Date formatting -->
  <p>{{ '{{ currentDate | date:\'fullDate\' }}' }}</p>
